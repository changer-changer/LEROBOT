```python
def init_keyboard_listener():
    events = {
        "exit_early": False,       # 停止当前 episode 录制
        "stop_recording": False,   # 完全退出程序
        "start_episode": False,    # S 键触发
        "discard_episode": False,  # Backspace/L 键触发
    }
    
    def on_press(key):
        if key.char == 's' or key.char == 'S':
            events["start_episode"] = True
        elif key == keyboard.Key.space:
            events["exit_early"] = True  # Space 保存
        elif key == keyboard.Key.backspace:
            events["discard_episode"] = True  # Backspace 丢弃
            events["exit_early"] = True
```

### 2.3 录制循环详解

```python
def manual_record_loop(dataset, events, fps, ...):
    """核心录制循环 - 每秒采集 fps 帧"""
    
    while timestamp < control_time_s:
        # 1. 检查退出信号
        if events["exit_early"]:
            events["exit_early"] = False
            break
        
        # 2. 获取观测
        obs = robot.get_observation()
        
        # 3. 构建数据帧
        frame = {
            **observation_frame,
            **action_frame,
            "task": single_task
        }
        
        # 4. 添加到 buffer
        dataset.add_frame(frame)
        
        # 5. 精确时间控制（关键！）
        dt_s = time.perf_counter() - start_loop_t
        sleep_time_s = max(0, 1 / fps - dt_s)
        busy_wait(sleep_time_s)  # 或 precise_sleep
```

### 3.3 Feature 类型详解

| dtype | 用途 | shape | 存储位置 |
|-------|------|-------|----------|
| `float32` | 连续值数据 (关节角度、力、点云) | 任意 | `data/*.parquet` |
| `int64` | 离散索引 (episode_index, frame_index) | `(1,)` | `data/*.parquet` |
| `video` | 视频数据 | `(H, W, C)` | `videos/*.mp4` |
| `image` | 图像数据 (PNG序列) | `(H, W, C)` | `images/*/*.png` |

**命名约定**：
- `action.*` - 动作数据 (机器人输入)
- `observation.state` - 状态观测 (位置/速度/力矩)
- `observation.images.{key}` - 图像观测
- `observation.{sensor}` - 其他传感器数据

```python
    # ... 其他 features
}
```

### 3.5 数据写入流程

```
1. add_frame(frame)
   ├── 验证 frame 格式 (validate_frame)
   ├── 自动添加 frame_index, timestamp
   ├── 图像数据 → 写入 PNG 临时文件
   ├── 视频数据 → 送入 StreamingEncoder (如果启用)
   └── 其他数据 → 存入 episode_buffer 列表

2. save_episode()
   ├── 验证 buffer
   ├── 分配全局 index
   ├── 视频编码 (如果 streaming_encoding=false)
   ├── 计算统计信息 (stats)
   ├── 保存到 parquet
   ├── 更新 episodes 元数据
   └── 清空 episode_buffer
```

```python
def warmup_cameras(robot, duration_s: float = 1.5) -> None:
    """相机预热，稳定自动曝光/白平衡"""
    cams = getattr(robot, "cameras", None)
    if not cams:
        return
    
    # 预热循环...
    while time.perf_counter() - start < duration_s:
        for cam in cam_list:

                if hasattr(cam, "async_read"):
                    cam.async_read(timeout_ms=500)
                else:
                    cam.read()
            except Exception:
                continue
```

### 4.4 视频质量问题

#### 问题 7：streaming_encoding 导致视频模糊
**现象**：MP4 文件非常模糊

**原因分析**：

| 编码模式 | 参数 | 质量 | 速度 |
|---------|------|------|------|
| 事后编码 | `streaming_encoding=false` | ⭐⭐⭐ 高 | 保存时慢 |
| 实时编码 | `streaming_encoding=true` | ⭐⭐ 中 | 保存时快 |

**解决方案**：
1. 移除 `--dataset.streaming_encoding=true` 参数
2. 使用默认的 `libsvtav1` 编码器（质量最好）
3. 或显式设置为 `false`：
   ```bash
   --dataset.streaming_encoding=false
   ```

#### 问题 8：128x128 强制 resize
**现象**：画面变形，变成正方形

**原因**：`ROS2CameraConfig` 默认 `width=128, height=128`

**解决方案**：
```python
# configs.py - 修改为默认保持原始分辨率
width: int | None = None
height: int | None = None
```

**命令行用法**：
```bash
# 保持原始分辨率
--robot.cameras='{"left_rgb": {"type": "ros2", "topic": "...", "fps": 30}}'

# 或指定分辨率
--robot.cameras='{"left_rgb": {"type": "ros2", "topic": "...", "fps": 30, "width": 640, "height": 480}}'
```

## 五、关键代码模板

### 5.1 完整的手动录制脚本模板

```python
#!/usr/bin/env python
"""
手动录制脚本模板
- S: 开始
- Space: 保存
- Backspace: 丢弃
"""

import logging
import time
from dataclasses import dataclass, field
from pathlib import Path

from lerobot.configs import parser
from lerobot.datasets.lerobot_dataset import LeRobotDataset
from lerobot.datasets.pipeline_features import (
    aggregate_pipeline_dataset_features,
    create_initial_features,
)
from lerobot.datasets.utils import combine_feature_dicts
from lerobot.datasets.video_utils import VideoEncodingManager
from lerobot.processor import make_default_processors
from lerobot.robots import make_robot_from_config
from lerobot.teleoperators import make_teleoperator_from_config
from lerobot.utils.control_utils import (
    init_keyboard_listener,
    is_headless,
)
from lerobot.utils.import_utils import register_third_party_plugins
from lerobot.utils.robot_utils import busy_wait
from lerobot.utils.utils import init_logging, log_say


@dataclass
class DatasetRecordConfig:
    """数据集配置"""
    repo_id: str
    single_task: str
    fps: int = 30
    episode_time_s: float = 60
    num_episodes: int = 50
    video: bool = True
    push_to_hub: bool = False
    vcodec: str = "libsvtav1"
    streaming_encoding: bool = False  # 默认高质量
    root: str | Path | None = None


@dataclass
class RecordConfig:
    """录制配置"""
    robot: RobotConfig
    dataset: DatasetRecordConfig
    teleop: TeleoperatorConfig | None = None
    display_data: bool = False


def record_loop(robot, dataset, events, fps, ...):
    """录制循环"""
    start_time = time.perf_counter()
    
    while True:
        # 检查退出
        if events.get("exit_early", False):
            events["exit_early"] = False
            break
        
        # 采集数据
        obs = robot.get_observation()
        frame = build_frame(obs, ...)
        dataset.add_frame(frame)
        
        # 控制帧率
        elapsed = time.perf_counter() - start_time
        busy_wait(max(0, 1/fps - elapsed))
        start_time = time.perf_counter()


@parser.wrap()
def record(cfg: RecordConfig):
    """主函数"""
    init_logging()
    
    # 创建组件
    robot = make_robot_from_config(cfg.robot)
    teleop = make_teleoperator_from_config(cfg.teleop) if cfg.teleop else None
    
    # 创建数据集
    dataset = LeRobotDataset.create(
        cfg.dataset.repo_id,
        cfg.dataset.fps,
        root=cfg.dataset.root,
        robot_type=robot.name,
        features=dataset_features,
        use_videos=cfg.dataset.video,
        vcodec=cfg.dataset.vcodec,
        streaming_encoding=cfg.dataset.streaming_encoding,
    )
    
    # 键盘监听
    listener, events = init_keyboard_listener()
    
    try:
        # 连接设备
        robot.connect()
        if teleop:
            teleop.connect()
        
        # 主循环
        with VideoEncodingManager(dataset):
            while dataset.num_episodes < cfg.dataset.num_episodes:
                # 等待开始
                print("按 'S' 开始录制...")
                while not events["start_episode"]:
                    time.sleep(0.05)
                events["start_episode"] = False
                
                # 录制
                print("录制中... (Space 保存, Backspace 丢弃)")
                record_loop(robot, dataset, events, cfg.dataset.fps, ...)
                
                # 保存或丢弃
                if events.get("discard_episode", False):
                    print("丢弃 episode...")
                    dataset.clear_episode_buffer()
                    events["discard_episode"] = False
                else:
                    print("保存 episode...")
                    dataset.save_episode()
                    
    finally:
        # 清理
        if listener and not is_headless():
            listener.stop()
        robot.disconnect()
        if teleop:
            teleop.disconnect()


def main():
    register_third_party_plugins()
    record()


if __name__ == "__main__":
    main()
```

## 八、Replay 回放脚本原理与开发

> 本节详细讲解如何将录制的数据集动作序列回放到机器人上，包括数据加载、动作执行、安全检查和帧率控制。

### 8.1 回放的核心原理

#### 8.1.1 什么是回放？

回放（Replay）是将录制好的数据集 `action` 序列重新发送给机器人执行的过程。与录制相反：

- **录制**: 机器人状态 → 数据集
- **回放**: 数据集动作 → 机器人执行

#### 8.1.2 数据流向

```
┌─────────────────────────────────────────────────────────────────┐
│                        回放流程                                   │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  [Dataset]          [Action Processor]         [Robot]           │
│      │                       │                    │              │
│      ▼                       ▼                    ▼              │
│  ┌──────────┐         ┌──────────┐         ┌──────────┐         │
│  │Load      │         │Safety    │         │Execute   │         │
│  │Episode   │────────▶│Check     │────────▶│Action    │         │
│  │          │         │& Smooth  │         │          │         │
│  └──────────┘         └──────────┘         └──────────┘         │
│       │                                              │          │
│       │                    [Keyboard]                │          │
│       │                       │                      │          │
│       ▼                       ▼                      ▼          │
│  ┌──────────┐         ┌──────────┐         ┌──────────┐         │
│  │Frame N+1 │         │Pause/   │         │Get Obs   │         │
│  └──────────┘         │Stop     │         │(Visual)  │         │
│                       └──────────┘         └──────────┘         │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

### 8.3 帧率控制

#### 8.3.1 为什么需要帧率控制？

如果不控制帧率，动作会尽可能快地发送，导致：
- 机器人运动过快，可能损坏
- 与录制时的时序不一致
- 无法预测的运动行为

#### 8.3.2 精确的帧率控制实现

```python
import time

def replay_with_fps_control(robot, actions, fps=30):
    """带帧率控制的回放"""
    frame_duration = 1.0 / fps  # 每帧应耗时 (33.33ms for 30fps)
    
    for idx in range(len(actions)):
        start_time = time.perf_counter()
        
        # 1. 获取并发送动作
        action = get_action(actions, idx)
        robot.send_action(action)
        
        # 2. 计算已经用时
        elapsed = time.perf_counter() - start_time
        
        # 3. 等待剩余时间
        sleep_time = frame_duration - elapsed
        if sleep_time > 0:
            busy_wait(sleep_time)  # 使用 busy_wait 而不是 time.sleep
```

**busy_wait vs time.sleep**:

| 方法 | 精度 | CPU 占用 | 适用场景 |
|------|------|---------|---------|
| `time.sleep()` | ~10ms 误差 | 低 | 长时间等待 |
| `busy_wait()` | ~0.1ms 误差 | 高 | 短时间、高精度 |

**busy_wait 实现原理**:
```python
def busy_wait(seconds):
    """忙等待，精度更高"""
    end_time = time.perf_counter() + seconds
    while time.perf_counter() < end_time:
        pass  # 空循环，占用 CPU

# 改进版：混合使用 sleep + spin
def precise_sleep(seconds, spin_threshold=0.010):
    """先 sleep 大部分时间，最后 spin 精准对齐"""
    if seconds <= 0:
        return
    
    end_time = time.perf_counter() + seconds
    
    # 如果剩余时间 > 10ms，先 sleep
    if seconds > spin_threshold:
        time.sleep(seconds - spin_threshold)
    
    # 最后 10ms 用忙等待精准对齐
    while time.perf_counter() < end_time:
        pass
```

### 8.4 安全检查机制

#### 8.4.1 为什么需要安全检查？

数据集可能包含危险动作：
- 超出关节限位的目标值
- 速度过快的动作变化
- 损坏或错误的数据

#### 8.4.2 安全检查实现

```python
def check_action_safety(action: dict, robot, max_speed: float = 1.0):
    """
    检查动作安全性
    
    Returns:
        (is_safe, message)
    """
    # 1. 获取当前状态
    current_obs = robot.get_observation()
    current_q = np.array([
        current_obs[f"joint_{i}_pos"] for i in range(16)
    ])
    
    # 2. 构建目标位置
    target_q = np.zeros(16)
    for i in range(16):
        key = f"action.joint_{i}_pos"
        target_q[i] = action.get(key, current_q[i])
    
    # 3. 检查关节限位
    joint_limits_low = np.array([
        -3.5, -0.5, -4.0, -3.0, -2.0, -1.0, -1.8,  # 左臂
        -3.5, -0.5, -4.0, -3.0, -2.0, -1.0, -1.8,  # 右臂
        -5, -5  # 夹爪
    ])
    joint_limits_high = np.array([
        3.0, 3.2, 1.6, 0.6, 1.5, 0.9, 1.8,   # 左臂
        3.0, 3.2, 1.6, 0.6, 1.5, 0.9, 1.8,   # 右臂
        105, 105  # 夹爪
    ])
    
    for i in range(14):  # 只检查臂部
        if target_q[i] < joint_limits_low[i] or target_q[i] > joint_limits_high[i]:
            return False, f"关节 {i} 超出限位"
    
    # 4. 检查速度
    joint_speed = np.abs(target_q - current_q) * 30  # 假设 30fps
    if np.max(joint_speed[:14]) > max_speed * 2:
        return False, f"速度超过限制"
    
    return True, "OK"
```

### 8.5 动作平滑

#### 8.5.1 为什么需要平滑？

录制时的动作可能有噪声或抖动：
- 人手抖动
- 传感器噪声
- 快速的方向改变

#### 8.5.2 指数平滑算法

```python
def smooth_action(prev_action: dict, target_action: dict, alpha: float):
    """
    指数平滑
    
    alpha: 平滑系数 (0-1)
        - 0: 完全使用上一帧（极度平滑，延迟大）
        - 1: 完全使用当前帧（无平滑）
        - 0.3: 30% 当前帧 + 70% 上一帧（推荐）
    """
    smoothed = {}
    for key in target_action:
        if key in prev_action:
            smoothed[key] = alpha * target_action[key] + (1 - alpha) * prev_action[key]
        else:
            smoothed[key] = target_action[key]
    return smoothed

# 使用示例
prev_action = None
for idx in range(total_frames):
    action = get_action(actions, idx)
    
    if action_smoothing > 0 and prev_action is not None:
        action = smooth_action(prev_action, action, action_smoothing)
    
    prev_action = action.copy()
    robot.send_action(action)
```

**平滑效果对比** (假设连续 5 帧的关节 0 目标值):

| 帧 | 原始值 | alpha=0.0 | alpha=0.3 | alpha=0.7 |
|:--:|:------:|:---------:|:---------:|:---------:|
| 1 | 0.5 | 0.5 | 0.5 | 0.5 |
| 2 | 0.7 | 0.5 | 0.56 | 0.64 |
| 3 | 0.3 | 0.5 | 0.44 | 0.35 |
| 4 | 0.8 | 0.5 | 0.55 | 0.67 |
| 5 | 0.6 | 0.5 | 0.56 | 0.62 |

### 8.6 键盘控制实现

#### 8.6.1 键盘事件监听

```python
from pynput import keyboard

def init_keyboard_listener():
    """初始化键盘监听器"""
    events = {
        "pause": False,      # 暂停状态
        "stop": False,       # 停止回放
        "step_next": False,  # 单步：下一步
    }
    
    def on_press(key):
        try:
            if key == keyboard.Key.space:
                # Space: 切换暂停/继续
                events["pause"] = not events["pause"]
                state = "PAUSED" if events["pause"] else "RESUMED"
                print(f"Playback {state}")
                
            elif key == keyboard.Key.esc:
                # ESC: 停止
                events["stop"] = True
                print("Playback STOP requested")
                
            elif hasattr(key, 'char') and key.char == 'n':
                # N: 单步执行
                events["step_next"] = True
                
        except AttributeError:
            pass
    
    listener = keyboard.Listener(on_press=on_press)
    listener.start()
    return listener, events
```

### 8.7 完整的回放循环

```python
def replay_episode(robot, dataset, episode, fps, events, **kwargs):
    """完整的 episode 回放函数"""
    
    # 1. 加载 episode 数据
    episode_frames = dataset.hf_dataset.filter(
        lambda x: x["episode_index"] == episode
    )
    actions = episode_frames.select_columns("action")
    action_names = dataset.features["action"]["names"]
    total_frames = len(episode_frames)
    
    # 2. 初始化
    prev_action = None
    frame_duration = 1.0 / fps
    
    # 3. 回放循环
    for idx in range(total_frames):
        start_time = time.perf_counter()
        
        # 3.1 检查停止信号
        if events["stop"]:
            break
        
        # 3.2 处理暂停
        while events["pause"] and not events["stop"]:
            time.sleep(0.05)
            if events["step_next"]:
                events["step_next"] = False
                break
        
        # 3.3 构建动作
        action_array = actions[idx]["action"]
        action = {name: float(action_array[i]) 
                  for i, name in enumerate(action_names)}
        
        # 3.4 动作平滑
        if kwargs.get("action_smoothing", 0) > 0 and prev_action:
            action = smooth_action(prev_action, action, 
                                   kwargs["action_smoothing"])
        prev_action = action.copy()
        
        # 3.5 安全检查
        if kwargs.get("enable_safety_check", True):
            is_safe, msg = check_action_safety(action, robot, 
                                               kwargs.get("max_speed", 1.0))
            if not is_safe:
                logging.error(f"Safety check failed: {msg}")
                break
        
        # 3.6 发送动作
        try:
            robot.send_action(action)
        except Exception as e:
            logging.error(f"Send action failed: {e}")
            break
        
        # 3.7 帧率控制
        elapsed = time.perf_counter() - start_time
        sleep_time = frame_duration - elapsed
        if sleep_time > 0:
            busy_wait(sleep_time)
```

## 附录：MoveJ vs ServoJ 控制方式对比

### A.1 ServoJ（实时伺服）

**原理**：
- 高频发送目标关节位置（50-100Hz）
- 机器人实时跟踪目标位置
- 需要输入数据非常平滑

**代码实现**：
```python
# 每 20ms 发送一次目标位置
while True:
    target = get_target()  # 获取目标位置
    robot.send_action({"action.joint_0_pos": target})
    time.sleep(0.02)  # 50Hz
```

**问题**：
- 如果相邻两帧目标位置差距大 → 机器人剧烈加速 → **振荡**
- 对数据质量要求高

---

### A.2 MoveJ（关节插值）

**原理**：
- 发送目标位置 + 运动时间
- 机器人内部自动规划平滑轨迹
- 使用多项式插值（通常 3-5 次）

**代码实现**：
```python
# 发送一次，机器人自己规划轨迹
target = get_target()
robot.send_action_movej(
    {"action.joint_0_pos": target},
    move_time=0.1  # 0.1秒内平滑到达
)
```

**优势**：
- 自动保证速度和加速度连续
- 不会出现振荡
- 适合回放预录制的轨迹

---

### A.3 什么时候用什么？

| 场景 | 推荐方法 | 原因 |
|------|---------|------|
| **回放数据集** | MoveJ | 数据集可能不平滑，MoveJ 自动插值 |
| **VR 遥操作** | ServoJ | 实时性要求高，操作者自己控制平滑度 |
| ** policy 推理** | MoveJ | 策略输出可能有噪声，MoveJ 滤波 |
| **示教编程** | MoveJ | 关键点位之间自动插值 |

### A.4 MoveJ 的轨迹规划

MoveJ 内部通常使用**三次或五次多项式插值**：

```
θ(t) = a₀ + a₁t + a₂t² + a₃t³  (三次多项式)

约束条件：
- θ(0) = θ_start      (起点位置)
- θ(T) = θ_end        (终点位置)  
- θ'(0) = v_start     (起点速度，通常 0)
- θ'(T) = v_end       (终点速度，通常 0)
```

这样可以保证：
- 位置连续 ✓
- 速度连续 ✓  
- 加速度连续 ✓（五次多项式）

因此不会出现 ServoJ 的"突变→振荡"问题。
